2. 指定目录内寻找指定文件

  • 输入目录与文件名，该目录中搜索该文件，获得所有该文件的路径。

        @return array

function getFilePath($dir, $file, &$path = array()) {
    $dir_handle = opendir($dir);

    while ($file = readdir($dir_handle)) {
        if ($file == '.' || $file == '..') {
            continue;
        }
        $dirFile = $dir . '/' . $file;
        if (is_dir($dirFile)) {
            getFilePath($dirFile, $file, $path);
        } elseif (basename($dirFile) == $file) {
            $path[] = $dirFile;
        }
    }
    closedir($dir_handle);
    return $path;
}
